18def serialize(data):
19 """Serialize a dict into a JSON formatted string.
20
21 This function enforces rules like the separator and order of keys.
22 This ensures that all dicts are serialized in the same way.
23
24 This is specially important for hashing data. We need to make sure that
25 everyone serializes their data in the same way so that we do not have
26 hash mismatches for the same structure due to serialization
27 differences.
28
29 Args:
30 data (dict): dict to serialize
31
32 Returns:
33 str: JSON formatted string
34
35 """
36 return rapidjson.dumps(data, skipkeys=False, ensure_ascii=False,
37 sort_keys=True)
