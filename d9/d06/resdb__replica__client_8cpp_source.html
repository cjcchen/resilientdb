36ResDBReplicaClient::ResDBReplicaClient(const std::vector<ReplicaInfo>& replicas,
37 SignatureVerifier* verifier,
38 bool is_use_long_conn, int epoll_num,
39 int tcp_batch)
40 : replicas_(replicas),
41 verifier_(verifier),
42 is_running_(false),
43 batch_queue_("bc_batch", tcp_batch),
44 is_use_long_conn_(is_use_long_conn) {
45 global_stats_ = Stats::GetGlobalStats();
46 if (is_use_long_conn_) {
47 worker_ = std::make_unique<boost::asio::io_service::work>(io_service_);
48 for (int i = 0; i < epoll_num; ++i) {
49 worker_threads_.push_back(std::thread([&]() { io_service_.run(); }));
50 }
51 }
52
53 for (const ReplicaInfo& info : replicas) {
54 std::string ip = info.ip();
55 int port = info.port();
56 auto client = std::make_unique<AsyncReplicaClient>(
57 &io_service_, ip, port + (is_use_long_conn_ ? 10000 : 0), true);
58 client_pools_[std::make_pair(ip, port)] = std::move(client);
59 }
60
61 StartBroadcastInBackGround();
62}

64ResDBReplicaClient::~ResDBReplicaClient() {
65 is_running_ = false;
66 if (broadcast_thread_.joinable()) {
67 broadcast_thread_.join();
68 }
69 if (is_use_long_conn_) {
70 for (auto& cli : client_pools_) {
71 cli.second.reset();
72 }
73 worker_.reset();
74 worker_ = nullptr;
75 io_service_.stop();
76 for (auto& worker_th : worker_threads_) {
77 if (worker_th.joinable()) {
78 worker_th.join();
79 }
80 }
81 }
82}

116void ResDBReplicaClient::StartBroadcastInBackGround() {
117 is_running_ = true;
118 broadcast_thread_ = std::thread([&]() {
119 while (IsRunning()) {
120 std::vector<std::unique_ptr<QueueItem>> batch_req =
121 batch_queue_.Pop(10000);
122 if (batch_req.empty()) {
123 continue;
124 }
125 BroadcastData broadcast_data;
126 for (auto& queue_item : batch_req) {
127 broadcast_data.add_data()->swap(queue_item->data);
128 }
129
130 global_stats_->SendBroadCastMsg(broadcast_data.data_size());
131 int ret = SendMessageFromPool(broadcast_data, replicas_);
132 if (ret < 0) {
133 LOG(ERROR) << "broadcast request fail:";
134 }
135 }
136 });
137}

164int ResDBReplicaClient::SendBatchMessage(
165 const std::vector<std::unique_ptr<Request>>& messages,
166 const ReplicaInfo& replica_info) {
167 if (is_use_long_conn_) {
168 BroadcastData broadcast_data;
169 for (const auto& message : messages) {

171 broadcast_data.add_data()->swap(data);
172 }
173 return SendMessageFromPool(broadcast_data, {replica_info});
174 } else {
175 int ret = 0;
176 for (const auto& message : messages) {
177 ret += SendMessageInternal(*message, {replica_info});

183int ResDBReplicaClient::SendMessageFromPool(
184 const google::protobuf::Message& message,
185 const std::vector<ReplicaInfo>& replicas) {
186 int ret = 0;
187 std::string data;
188 message.SerializeToString(&data);
189 global_stats_->SendBroadCastMsgPerRep();
190 std::lock_guard<std::mutex> lk(mutex_);
191 for (const auto& replica : replicas) {
192 auto client = GetClientFromPool(replica.ip(), replica.port());
193 if (client == nullptr) {
194 continue;
195 }
196 if (client->SendMessage(data) == 0) {
197 ret++;
198 } else {
199 LOG(ERROR) << "send to:" << replica.ip() << " fail";
200 }
201 }
202 return ret;
203}

205int ResDBReplicaClient::SendMessageInternal(
206 const google::protobuf::Message& message,
207 const std::vector<ReplicaInfo>& replicas) {
208 int ret = 0;
209 for (const auto& replica : replicas) {
210 auto client = GetClient(replica.ip(), replica.port());
211 if (client == nullptr) {
212 continue;
213 }
214 if (verifier_ != nullptr) {
215 client->SetSignatureVerifier(verifier_);
216 }
217 if (client->SendRawMessage(message) == 0) {
218 ret++;
219 }
220 }
221 return ret;

224AsyncReplicaClient* ResDBReplicaClient::GetClientFromPool(const std::string& ip,
225 int port) {
226 if (client_pools_.find(std::make_pair(ip, port)) == client_pools_.end()) {
227 auto client = std::make_unique<AsyncReplicaClient>(
228 &io_service_, ip, port + (is_use_long_conn_ ? 10000 : 0), true);
229 client_pools_[std::make_pair(ip, port)] = std::move(client);
230 }

246void ResDBReplicaClient::SendMessage(const google::protobuf::Message& message,
247 int64_t node_id) {
248 ReplicaInfo target_replica;
249 for (const auto& replica : replicas_) {
250 if (replica.id() == node_id) {
251 target_replica = replica;
252 break;
253 }
254 }
255 if (target_replica.ip().empty()) {
256 for (const auto& replica : GetClientReplicas()) {
257 if (replica.id() == node_id) {
258 target_replica = replica;
259 break;
260 }
261 }
262 }
263
264 if (target_replica.ip().empty()) {
265 LOG(ERROR) << "no replica info";
266 return;
267 }
268
269 int ret = SendMessage(message, target_replica);
270 if (ret < 0) {
271 LOG(ERROR) << "broadcast request fail:";
272 }
273}
