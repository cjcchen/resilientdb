43absl::StatusOr<std::string> ResDBUserClient::GetResponseData(
44 const Response& response) {
45 std::string hash_;
46 std::set<int64_t> hash_counter;
47 std::string resp_str;
48 for (const auto& each_resp : response.resp()) {
49 // Check signature
50 std::string hash = SignatureVerifier::CalculateHash(each_resp.data());
51
52 if (!hash_.empty() && hash != hash_) {
53 LOG(ERROR) << "hash not the same";
54 return absl::InvalidArgumentError("hash not match");
55 }
56 if (hash_.empty()) {
57 hash_ = hash;
58 resp_str = each_resp.data();
59 }
60 hash_counter.insert(each_resp.signature().node_id());
61 }
62 // LOG(INFO) << "recv hash:" << hash_counter.size()
63 // << " need:" << config_.GetMinClientReceiveNum()
64 // << " data len:" << resp_str.size();
65 if (hash_counter.size() >=
66 static_cast<size_t>(config_.GetMinClientReceiveNum())) {
67 return resp_str;
68 }
69 return absl::InvalidArgumentError("data not enough");
70}

79int ResDBUserClient::SendRequest(const google::protobuf::Message& message,
80 google::protobuf::Message* response,
81 Request::Type type) {
82 ResDBClient::SetDestReplicaInfo(config_.GetReplicaInfos()[0]);
83 int ret = ResDBClient::SendRequest(message, type, true);
84 if (ret == 0) {
85 std::string resp_str;
86 int ret = ResDBClient::RecvRawMessageData(&resp_str);
87 if (ret >= 0) {
88 if (!response->ParseFromString(resp_str)) {
89 LOG(ERROR) << "parse response fail:" << resp_str.size();
90 return -2;
91 }
92 return 0;
93 }
94 }
95 return -1;
96}
