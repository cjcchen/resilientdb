63// Handle the client request and send a pre-prepare message to others.
64// TODO if not a primary, redicet to the primary replica.
65int Commitment::ProcessNewRequest(std::unique_ptr<Context> context,
66 std::unique_ptr<Request> client_request) {
67 if (context == nullptr || context->signature.signature().empty()) {
68 LOG(ERROR) << "client request doesn't contain signature, reject";
69 return -2;
70 }
71
72 if (config_.GetSelfInfo().id() != transaction_manager_->GetCurrentPrimary()) {
73 LOG(ERROR) << "current node is not primary. primary:"
74 << transaction_manager_->GetCurrentPrimary()
75 << " seq:" << client_request->seq();
76 return -2;
77 }
78
79 if (pre_verify_func_ && !pre_verify_func_(*client_request)) {
80 LOG(ERROR) << " check by the user func fail";
81 return -2;
82 }
83
84 global_stats_->IncClientRequest();
85 auto seq = transaction_manager_->AssignNextSeq();
86 if (!seq.ok()) {
87 global_stats_->SeqFail();
88 Request request;
89 request.set_type(Request::TYPE_RESPONSE);
90 request.set_sender_id(config_.GetSelfInfo().id());
91 request.set_proxy_id(client_request->proxy_id());
92 request.set_ret(-2);
93
94 replica_client_->SendMessage(request, request.proxy_id());
95 return -2;
96 }
97
98 client_request->set_type(Request::TYPE_PRE_PREPARE);
99 client_request->set_current_view(transaction_manager_->GetCurrentView());
100 client_request->set_seq(*seq);
101 client_request->set_sender_id(config_.GetSelfInfo().id());
102 replica_client_->BroadCast(*client_request);
103 return 0;
104}

106// Receive the pre-prepare message from the primary.
107// TODO check whether the sender is the primary.
108int Commitment::ProcessProposeMsg(std::unique_ptr<Context> context,
109 std::unique_ptr<Request> request) {
110 if (context == nullptr || context->signature.signature().empty()) {
111 LOG(ERROR) << "client request doesn't contain signature, reject";
112 return -2;
113 }
114
115 if (request->sender_id() != transaction_manager_->GetCurrentPrimary()) {
116 LOG(ERROR) << "the request is not from primary. sender:"
117 << request->sender_id() << " seq:" << request->seq();
118 return -2;
119 }
120
121 if (request->sender_id() != config_.GetSelfInfo().id()) {
122 if (pre_verify_func_ && !pre_verify_func_(*request)) {
123 LOG(ERROR) << " check by the user func fail";
124 return -2;
125 }
126 BatchClientRequest batch_request;
127 batch_request.ParseFromString(request->data());
128 batch_request.clear_createtime();
129 std::string data;
130 batch_request.SerializeToString(&data);
131 // check signatures
132 bool valid =
133 verifier_->VerifyMessage(request->data(), request->data_signature());
134 if (!valid) {
135 LOG(ERROR) << "request is not valid:"
136 << request->data_signature().DebugString();
137 LOG(ERROR) << " msg:" << request->data().size();
138 return -2;
139 }
140 }
141
142 global_stats_->IncPropose();
143 std::unique_ptr<Request> prepare_request = resdb::NewRequest(
144 Request::TYPE_PREPARE, *request, config_.GetSelfInfo().id());
145 prepare_request->clear_data();
146
147 // Add request to message_manager.
148 // If it has received enough same requests(2f+1), broadcast the prepare
149 // message.
150 CollectorResultCode ret = transaction_manager_->AddConsensusMsg(
151 context->signature, std::move(request));
152 if (ret == CollectorResultCode::STATE_CHANGED) {
153 replica_client_->BroadCast(*prepare_request);
154 }
155 return ret == CollectorResultCode::INVALID ? -2 : 0;
156}

158// If receive 2f+1 prepare message, broadcast a commit message.
159int Commitment::ProcessPrepareMsg(std::unique_ptr<Context> context,
160 std::unique_ptr<Request> request) {
161 if (context == nullptr || context->signature.signature().empty()) {
162 LOG(ERROR) << "client request doesn't contain signature, reject";
163 return -2;
164 }
165 global_stats_->IncPrepare();
166 request->mutable_data_signature()->Clear();
167 std::unique_ptr<Request> commit_request = resdb::NewRequest(
168 Request::TYPE_COMMIT, *request, config_.GetSelfInfo().id());
169 // Add request to message_manager.
170 // If it has received enough same requests(2f+1), broadcast the commit
171 // message.
172 CollectorResultCode ret = transaction_manager_->AddConsensusMsg(
173 context->signature, std::move(request));
174 if (ret == CollectorResultCode::STATE_CHANGED) {
175 // If need qc, sign the data
176 if (need_qc_ && verifier_) {
177 auto signature_or = verifier_->SignMessage(commit_request->hash());
178 if (!signature_or.ok()) {
179 LOG(ERROR) << "Sign message fail";
180 return -2;
181 }
182 *commit_request->mutable_data_signature() = *signature_or;
183 }
184 replica_client_->BroadCast(*commit_request);
185 }
186 return ret == CollectorResultCode::INVALID ? -2 : 0;
187}

189// If receive 2f+1 commit message, commit the request.
190int Commitment::ProcessCommitMsg(std::unique_ptr<Context> context,
191 std::unique_ptr<Request> request) {
192 if (context == nullptr || context->signature.signature().empty()) {
193 LOG(ERROR) << "client request doesn't contain signature, reject";
194 return -2;
195 }
196 global_stats_->IncCommit();
197 // Add request to message_manager.
198 // If it has received enough same requests(2f+1), message manager will
199 // commit the request.
200 CollectorResultCode ret = transaction_manager_->AddConsensusMsg(
201 context->signature, std::move(request));
202 return ret == CollectorResultCode::INVALID ? -2 : 0;
203}

205// =========== private threads ===========================
206// If the transaction is executed, send back to the proxy.
207int Commitment::PostProcessExecutedMsg() {
208 while (!stop_) {
209 auto batch_resp = transaction_manager_->GetResponseMsg();
210 if (batch_resp == nullptr) {
211 continue;
212 }
213 Request request;
214 request.set_seq(batch_resp->seq());
215 request.set_type(Request::TYPE_RESPONSE);
216 request.set_sender_id(config_.GetSelfInfo().id());
217 request.set_current_view(batch_resp->current_view());
218 request.set_proxy_id(batch_resp->proxy_id());
219 batch_resp->SerializeToString(request.mutable_data());
220 replica_client_->SendMessage(request, request.proxy_id());
221 }
222 return 0;
223}
