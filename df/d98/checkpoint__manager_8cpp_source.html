83// check whether there are 2f+1 valid checkpoint proof.
84bool CheckPointManager::IsValidCheckpointProof(
85 const StableCheckPoint& stable_ckpt) {
86 std::string hash = stable_ckpt_.hash();
87 std::set<uint32_t> senders;
88 for (const auto& signature : stable_ckpt_.signatures()) {
89 if (!verifier_->VerifyMessage(hash, signature)) {
90 return false;
91 }
92 senders.insert(signature.node_id());
93 }
94 return (senders.size() >= config_.GetMinDataReceiveNum()) ||
95 (stable_ckpt.seq() == 0 && senders.size() == 0);
96}

98int CheckPointManager::ProcessCheckPoint(std::unique_ptr<Context> context,
99 std::unique_ptr<Request> request) {
100 CheckPointData checkpoint_data;
101 if (!checkpoint_data.ParseFromString(request->data())) {
102 LOG(ERROR) << "parse checkpont data fail:";
103 return -2;
104 }
105 uint64_t checkpoint_seq = checkpoint_data.seq();
106 uint32_t sender_id = request->sender_id();
107 int water_mark = config_.GetCheckPointWaterMark();
108 if (checkpoint_seq % water_mark) {
109 LOG(ERROR) << "checkpoint seq not invalid:" << checkpoint_seq;
110 return -2;
111 }
112
113 if (verifier_) {
114 // check signatures
115 bool valid = verifier_->VerifyMessage(checkpoint_data.hash(),
116 checkpoint_data.hash_signature());
117 if (!valid) {
118 LOG(ERROR) << "request is not valid:"
119 << checkpoint_data.hash_signature().DebugString();
120 return -2;
121 }
122 }
123
124 {
125 std::lock_guard<std::mutex> lk(mutex_);
126 auto res =
127 sender_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
128 .insert(sender_id);
129 if (res.second) {
130 sign_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
131 .push_back(checkpoint_data.hash_signature());
132 new_data_++;
133 }
134 Notify();
135 }
136 return 0;
137}

145 int timeout_ms = 1000;

151void CheckPointManager::UpdateStableCheckPointStatus() {
152 while (!stop_) {
153 if (!Wait()) {
154 continue;
155 }
156 uint64_t stable_seq = 0;
157 std::string stable_hash;
158 {
159 std::lock_guard<std::mutex> lk(mutex_);
160 for (auto it : sender_ckpt_) {
161 if (it.second.size() >=
162 static_cast<size_t>(config_.GetMinDataReceiveNum())) {
163 stable_seq = it.first.first;
164 stable_hash = it.first.second;
165 }
166 }
167 new_data_ = 0;
168 }
169
170 LOG(ERROR) << "current stable seq:" << current_stable_seq_
171 << " stable seq:" << stable_seq;
172 std::vector<SignatureInfo> votes;
173 if (current_stable_seq_ < stable_seq) {
174 std::lock_guard<std::mutex> lk(mutex_);
175 votes = sign_ckpt_[std::make_pair(stable_seq, stable_hash)];
176
177 auto it = sender_ckpt_.begin();
178 while (it != sender_ckpt_.end()) {
179 if (it->first.first <= stable_seq) {
180 sign_ckpt_.erase(sign_ckpt_.find(it->first));
181 auto tmp = it++;
182 sender_ckpt_.erase(tmp);
183 } else {
184 it++;
185 }
186 }
187 stable_ckpt_.set_seq(stable_seq);
188 stable_ckpt_.set_hash(stable_hash);
189 for (auto vote : votes) {
190 *stable_ckpt_.add_signatures() = vote;
191 }
192 current_stable_seq_ = stable_seq;
193 LOG(INFO) << "done. stable seq:" << current_stable_seq_
194 << " votes:" << stable_ckpt_.DebugString();
195 }
196 UpdateStableCheckPointCallback(current_stable_seq_);
197 }
198}

215void CheckPointManager::UpdateCheckPointStatus() {
216 uint64_t last_seq = 0;
217 uint64_t last_ckpt_seq = 0;
218 std::string last_hash;
219 int water_mark = config_.GetCheckPointWaterMark();
220 int timeout_ms = config_.GetViewchangeCommitTimeout();
221 while (!stop_) {
222 auto request = data_queue_.Pop(timeout_ms);
223 if (request == nullptr) {
224 TimeoutHandler();
225 continue;
226 }
227 uint64_t current_seq = request->seq();
228 if (current_seq != last_seq + 1) {
229 LOG(ERROR) << "seq invalid:" << last_seq << " current:" << current_seq;
230 continue;
231 }
232 last_hash = GetHash(last_hash, request->hash());
233 last_seq++;
234 txn_db_->Put(std::move(request));
235
236 if (current_seq == last_ckpt_seq + water_mark) {
237 last_ckpt_seq = current_seq;
238 BroadcastCheckPoint(last_ckpt_seq, last_hash);
239 }
240 }
241 return;
242}

244void CheckPointManager::BroadcastCheckPoint(uint64_t seq,
245 const std::string& hash) {
246 CheckPointData checkpoint_data;
247 std::unique_ptr<Request> checkpoint_request = NewRequest(
248 Request::TYPE_CHECKPOINT, Request(), config_.GetSelfInfo().id());
249 checkpoint_data.set_seq(seq);
250 checkpoint_data.set_hash(hash);
251 if (verifier_) {
252 auto signature_or = verifier_->SignMessage(hash);
253 if (!signature_or.ok()) {
254 LOG(ERROR) << "Sign message fail";
255 return;
256 }
257 *checkpoint_data.mutable_hash_signature() = *signature_or;
258 }
259
260 checkpoint_data.SerializeToString(checkpoint_request->mutable_data());
261 replica_client_->BroadCast(*checkpoint_request);
262}
