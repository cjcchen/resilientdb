100absl::StatusOr<std::pair<std::unique_ptr<Context>, std::unique_ptr<Request>>>
101ConsensusServicePBFT::PopPendingRequest() {
102 std::lock_guard<std::mutex> lk(mutex_);
103 if (request_pending_.empty()) {
104 LOG(ERROR) << "empty:";
105 return absl::InternalError("No Data.");
106 }
107 auto new_request = std::move(request_pending_.front());
108 request_pending_.pop();
109 return new_request;
110}

112// The implementation of PBFT.
113int ConsensusServicePBFT::ConsensusCommit(std::unique_ptr<Context> context,
114 std::unique_ptr<Request> request) {
115 // LOG(INFO) << "recv impl type:" << request->type() << " "
116 // << "sender id:" << request->sender_id();
117 // If it is in viewchange, push the request to the queue
118 // for the requests from the new view which come before
119 // the local new view done.
120 if (config_.GetConfigData().enable_viewchange()) {
121 view_change_manager_->MayStart();
122 if (view_change_manager_->IsInViewChange()) {
123 switch (request->type()) {
124 case Request::TYPE_NEW_TXNS:
125 case Request::TYPE_PRE_PREPARE:
126 case Request::TYPE_PREPARE:
127 case Request::TYPE_COMMIT:
128 AddPendingRequest(std::move(context), std::move(request));
129 return 0;
130 }
131 } else {
132 while (true) {
133 auto new_request = PopPendingRequest();
134 if (!new_request.ok()) {
135 break;
136 }
137 InternalConsensusCommit(std::move((*new_request).first),
138 std::move((*new_request).second));
139 }
140 }
141 }
142 return InternalConsensusCommit(std::move(context), std::move(request));
143}

145int ConsensusServicePBFT::InternalConsensusCommit(
146 std::unique_ptr<Context> context, std::unique_ptr<Request> request) {
147 // LOG(INFO) << "recv impl type:" << request->type() << " "
148 // << "sender id:" << request->sender_id();
149 switch (request->type()) {
150 case Request::TYPE_CLIENT_REQUEST:
151 if (config_.IsPerformanceRunning()) {
152 return performance_manager_->StartEval();
153 }
154 return response_manager_->NewClientRequest(std::move(context),
155 std::move(request));
156 case Request::TYPE_RESPONSE:
157 if (config_.IsPerformanceRunning()) {
158 return performance_manager_->ProcessResponseMsg(std::move(context),
159 std::move(request));
160 }
161 return response_manager_->ProcessResponseMsg(std::move(context),
162 std::move(request));
163 case Request::TYPE_NEW_TXNS:
164 return commitment_->ProcessNewRequest(std::move(context),
165 std::move(request));
166 case Request::TYPE_PRE_PREPARE:
167 return commitment_->ProcessProposeMsg(std::move(context),
168 std::move(request));
169 case Request::TYPE_PREPARE:
170 return commitment_->ProcessPrepareMsg(std::move(context),
171 std::move(request));
172 case Request::TYPE_COMMIT:
173 return commitment_->ProcessCommitMsg(std::move(context),
174 std::move(request));
175 case Request::TYPE_CHECKPOINT:
176 return checkpoint_manager_->ProcessCheckPoint(std::move(context),
177 std::move(request));
178 case Request::TYPE_VIEWCHANGE:
179 return view_change_manager_->ProcessViewChange(std::move(context),
180 std::move(request));
181 case Request::TYPE_NEWVIEW:
182 return view_change_manager_->ProcessNewView(std::move(context),
183 std::move(request));
184 case Request::TYPE_QUERY:
185 return query_->ProcessQuery(std::move(context), std::move(request));
186 case Request::TYPE_REPLICA_STATE:
187 return query_->ProcessGetReplicaState(std::move(context),
188 std::move(request));
189 case Request::TYPE_CUSTOM_QUERY:
190 return query_->ProcessCustomQuery(std::move(context), std::move(request));
191 }
192 return 0;
193}
