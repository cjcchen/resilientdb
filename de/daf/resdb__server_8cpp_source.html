77void ResDBServer::AcceptorHandler(const char* buffer, size_t data_len) {
78 BroadcastData data;
79 if (!data.ParseFromArray(buffer, data_len)) {
80 LOG(ERROR) << "parse broad cast fail:" << data_len;
81 return;
82 }
83
84 for (auto& sub_data : data.data()) {
85 std::unique_ptr<DataInfo> sub_request_info = std::make_unique<DataInfo>();
86 sub_request_info->data_len = sub_data.size();
87 sub_request_info->buff = new char[sub_request_info->data_len];
88 memcpy(sub_request_info->buff, sub_data.data(), sub_request_info->data_len);
89 std::unique_ptr<QueueItem> item = std::make_unique<QueueItem>();
90 item->socket = nullptr;
91 item->data = std::move(sub_request_info);
92 // LOG(ERROR) << "receve data from acceptor:" << data.is_resp()<<" data
93 // len:"<<item->data->data_len;
94 global_stats_->ServerCall();
95 input_queue_.Push(std::move(item));
96 }
97}

114 }));

233// Receive a message from network and pass it to service to process.
234void ResDBServer::Process(std::unique_ptr<QueueItem> item) {
235 auto client_socket =
236 item->socket == nullptr ? nullptr : std::move(item->socket);
237 auto request_info = std::move(item->data);
238 if (client_socket != nullptr) {
239 client_socket->SetSendTimeout(1000000);
240 }
241 std::unique_ptr<Context> context = std::make_unique<Context>();
242 context->client = std::make_unique<ResDBClient>(std::move(client_socket),
243 /*connected=*/true);
244 if (request_info) {
245 service_->Process(std::move(context), std::move(request_info));
246 }
247 return;
248}
