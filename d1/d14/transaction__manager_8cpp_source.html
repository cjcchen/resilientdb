83absl::StatusOr<uint64_t> TransactionManager::AssignNextSeq() {
84 std::unique_lock<std::mutex> lk(seq_mutex_);
85 uint32_t max_executed_seq = transaction_executor_->GetMaxPendingExecutedSeq();
86 global_stats_->SeqGap(next_seq_ - max_executed_seq);

88 static_cast<uint64_t>(config_.GetMaxProcessTxn())) {
89 return absl::InvalidArgumentError("Seq has been used up.");

98// Check if the request is valid.
99// 1. view is the same as the current view
100// 2. seq is larger or equal than the next execute seq.
101// 3. inside the water mark.
102bool TransactionManager::IsValidMsg(const Request& request) {
103 if (request.type() == Request::TYPE_RESPONSE) {
104 return true;
105 }
106 // view should be the same as the current one.
107 if (static_cast<uint64_t>(request.current_view()) != GetCurrentView()) {
108 LOG(ERROR) << "message view :[" << request.current_view()
109 << "] is older than the cur view :[" << GetCurrentView() << "]";
110 return false;
111 }
112
113 if (static_cast<uint64_t>(request.seq()) <
114 transaction_executor_->GetMaxPendingExecutedSeq()) {
115 return false;
116 }
117
118 return true;
119}

121bool TransactionManager::MayConsensusChangeStatus(
122 int type, int received_count, std::atomic<TransactionStatue>* status) {
123 switch (type) {
124 case Request::TYPE_PRE_PREPARE:
125 if (*status == TransactionStatue::None) {
126 TransactionStatue old_status = TransactionStatue::None;
127 return status->compare_exchange_strong(
128 old_status, TransactionStatue::READY_PREPARE,
129 std::memory_order_acq_rel, std::memory_order_acq_rel);
130 }
131 break;
132 case Request::TYPE_PREPARE:
133 if (*status == TransactionStatue::READY_PREPARE &&
134 config_.GetMinDataReceiveNum() <= received_count) {
135 TransactionStatue old_status = TransactionStatue::READY_PREPARE;
136 return status->compare_exchange_strong(
137 old_status, TransactionStatue::READY_COMMIT,
138 std::memory_order_acq_rel, std::memory_order_acq_rel);
139 }
140 break;
141 case Request::TYPE_COMMIT:
142 if (*status == TransactionStatue::READY_COMMIT &&
143 config_.GetMinDataReceiveNum() <= received_count) {
144 TransactionStatue old_status = TransactionStatue::READY_COMMIT;
145 return status->compare_exchange_strong(
146 old_status, TransactionStatue::READY_EXECUTE,
147 std::memory_order_acq_rel, std::memory_order_acq_rel);
148 return true;
149 }
150 break;
151 }
152 return false;
153}

155// Add commit messages and return the number of messages have been received.
156// The commit messages only include post(pre-prepare), prepare and commit
157// messages. Messages are handled by state (PREPARE,COMMIT,READY_EXECUTE).
158
159// If there are enough messages and the state is changed after adding the
160// message, return 1, otherwise return 0. Return -2 if the request is not valid.
161CollectorResultCode TransactionManager::AddConsensusMsg(
162 const SignatureInfo& signature, std::unique_ptr<Request> request) {
163 if (request == nullptr || !IsValidMsg(*request)) {
164 return CollectorResultCode::INVALID;
165 }
166 int type = request->type();
167 uint64_t seq = request->seq();
168 int resp_received_count = 0;
169
170 int ret = collector_pool_->GetCollector(seq)->AddRequest(
171 std::move(request), signature, type == Request::TYPE_PRE_PREPARE,
172 [&](const Request& request, int received_count,
173 TransactionCollector::CollectorDataType* data,
174 std::atomic<TransactionStatue>* status) {
175 if (MayConsensusChangeStatus(type, received_count, status)) {
176 resp_received_count = 1;
177 }
178 });
179 if (ret != 0) {
180 return CollectorResultCode::INVALID;
181 }
182 if (resp_received_count > 0) {
183 return CollectorResultCode::STATE_CHANGED;
184 }
185 return CollectorResultCode::OK;
186}

188RequestSet TransactionManager::GetRequestSet(uint64_t min_seq,
189 uint64_t max_seq) {
190 RequestSet ret;
191 std::unique_lock<std::mutex> lk(data_mutex_);
192 for (uint64_t i = min_seq; i <= max_seq; ++i) {
193 if (committed_data_.find(i) == committed_data_.end()) {
194 LOG(ERROR) << "seq :" << i << " doesn't exist";
195 continue;
196 }
197 RequestWithProof* request = ret.add_requests();
198 *request->mutable_request() = committed_data_[i];
199 request->set_seq(i);
200 for (const auto& request_info : committed_proof_[i]) {
201 RequestWithProof::RequestData* data = request->add_proofs();
202 *data->mutable_request() = *request_info->request;
203 *data->mutable_signature() = request_info->signature;
204 }
205 }
206 return ret;
207}
