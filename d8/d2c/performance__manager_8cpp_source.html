100// handle the response message. If receive f+1 commit messages, send back to the
101// client.
102int PerformanceManager::ProcessResponseMsg(std::unique_ptr<Context> context,
103 std::unique_ptr<Request> request) {
104 std::unique_ptr<Request> response;
105 // Add the response message, and use the call back to collect the received
106 // messages.
107 // The callback will be triggered if it received f+1 messages.
108 if (request->ret() == -2) {
109 // LOG(INFO) << "get response fail:" << request->ret();
110 send_num_--;
111 return 0;
112 }
113 CollectorResultCode ret =
114 AddResponseMsg(std::move(request), [&](const Request& request) {
115 response = std::make_unique<Request>(request);
116 return;
117 });
118
119 if (ret == CollectorResultCode::STATE_CHANGED) {
120 BatchClientResponse batch_response;
121 if (batch_response.ParseFromString(response->data())) {
122 SendResponseToClient(batch_response);
123 } else {
124 LOG(ERROR) << "parse response fail:";
125 }
126 }
127 return ret == CollectorResultCode::INVALID ? -2 : 0;
128}

130CollectorResultCode PerformanceManager::AddResponseMsg(
131 std::unique_ptr<Request> request,
132 std::function<void(const Request&)> response_call_back) {
133 if (request == nullptr) {
134 return CollectorResultCode::INVALID;
135 }
136
137 uint64_t seq = request->seq();
138
139 bool done = false;
140 {
141 int idx = seq % response_set_size_;
142 std::unique_lock<std::mutex> lk(response_lock_[idx]);
143 if (response_[idx][seq] == -1) {
144 return CollectorResultCode::OK;
145 }
146 response_[idx][seq]++;
147 if (response_[idx][seq] >= config_.GetMinClientReceiveNum()) {
148 response_[idx][seq] = -1;
149 done = true;
150 }
151 }
152 if (done) {
153 response_call_back(*request);
154 return CollectorResultCode::STATE_CHANGED;
155 }
156 return CollectorResultCode::OK;
157}

159void PerformanceManager::SendResponseToClient(
160 const BatchClientResponse& batch_response) {
161 uint64_t create_time = batch_response.createtime();
162 uint64_t local_id = batch_response.local_id();
163 if (create_time > 0) {
164 uint64_t run_time = get_sys_clock() - create_time;
165 global_stats_->AddLatency(run_time);
166 } else {
167 LOG(ERROR) << "seq:" << local_id << " no resp";
168 }
169 send_num_--;
170}

172// =================== request ========================
173int PerformanceManager::BatchProposeMsg() {
174 LOG(WARNING) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
175 << " batch num:" << config_.ClientBatchNum()
176 << " max txn:" << config_.GetMaxProcessTxn();
177 std::vector<std::unique_ptr<QueueItem>> batch_req;
178 eval_ready_future_.get();
179 while (!stop_) {
180 if (send_num_ > config_.GetMaxProcessTxn()) {
181 usleep(100000);
182 continue;
183 }
184 if (batch_req.size() < config_.ClientBatchNum()) {
185 std::unique_ptr<QueueItem> item =
186 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
187 if (item == nullptr) {
188 continue;
189 }
190 batch_req.push_back(std::move(item));
191 if (batch_req.size() < config_.ClientBatchNum()) {
192 continue;
193 }
194 }
195 int ret = DoBatch(batch_req);
196 batch_req.clear();
197 if (ret != 0) {
198 Response response;
199 response.set_result(Response::ERROR);
200 for (size_t i = 0; i < batch_req.size(); ++i) {
201 if (batch_req[i]->context && batch_req[i]->context->client) {
202 int ret = batch_req[i]->context->client->SendRawMessage(response);
203 if (ret) {
204 LOG(ERROR) << "send resp" << response.DebugString()
205 << " fail ret:" << ret;
206 }
207 }
208 }
209 }
210 }
211 return 0;
212}

214int PerformanceManager::DoBatch(
215 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
216 auto new_request =
217 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
218 if (new_request == nullptr) {
219 return -2;
220 }
221
222 BatchClientRequest batch_request;
223 for (size_t i = 0; i < batch_req.size(); ++i) {
224 BatchClientRequest::ClientRequest* req =
225 batch_request.add_client_requests();
226 *req->mutable_request() = *batch_req[i]->client_request.get();
227 if (batch_req[i]->context) {
228 *req->mutable_signature() = batch_req[i]->context->signature;
229 }
230 req->set_id(i);
231 }
232
233 batch_request.set_createtime(get_sys_clock());
234 batch_request.SerializeToString(new_request->mutable_data());
235 if (verifier_) {
236 auto signature_or = verifier_->SignMessage(new_request->data());
237 if (!signature_or.ok()) {
238 LOG(ERROR) << "Sign message fail";
239 return -2;
240 }
241 *new_request->mutable_data_signature() = *signature_or;
242 }
243
244 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
245 new_request->set_proxy_id(config_.GetSelfInfo().id());
246
247 replica_client_->SendMessage(*new_request, GetPrimary());
248 global_stats_->BroadCastMsg();
249 send_num_++;
250 if (total_num_++ == 1000000) {
251 stop_ = true;
252 LOG(WARNING) << "total num is done:" << total_num_;
253 }
254 if (total_num_ % 10000 == 0) {
255 LOG(WARNING) << "total num is :" << total_num_;
256 }
257 global_stats_->IncClientCall();
258 return 0;
259}
