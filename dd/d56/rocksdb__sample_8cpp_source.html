40using rocksdb::DB;
41using rocksdb::Options;
42using rocksdb::PinnableSlice;
43using rocksdb::ReadOptions;
44using rocksdb::WriteBatch;
45using rocksdb::WriteOptions;
46
47std::string kDBPath = "/tmp/rocksdb_simple_example";
48
49int main() {
50 DB* db;
51 Options options;
52 // Optimize RocksDB. This is the easiest way to get RocksDB to perform well
53 options.IncreaseParallelism();
54 options.OptimizeLevelStyleCompaction();
55 // create the DB if it's not already present
56 options.create_if_missing = true;
57
58 // open DB
59 rocksdb::Status s = DB::Open(options, kDBPath, &db);
60 assert(s.ok());
61
62 // Put key-value
63 s = db->Put(WriteOptions(), "key1", "value");
64 assert(s.ok());
65 std::string value;
66 // get value
67 s = db->Get(ReadOptions(), "key1", &value);
68 assert(s.ok());
69 assert(value == "value");
70
71 // atomically apply a set of updates
72 {
73 WriteBatch batch;
74 batch.Delete("key1");
75 batch.Put("key2", value);
76 s = db->Write(WriteOptions(), &batch);
77 }
78
79 s = db->Get(ReadOptions(), "key1", &value);
80 assert(s.IsNotFound());
81
82 db->Get(ReadOptions(), "key2", &value);
83 assert(value == "value");
84
85 {
86 PinnableSlice pinnable_val;
87 db->Get(ReadOptions(), db->DefaultColumnFamily(), "key2", &pinnable_val);
88 assert(pinnable_val == "value");
89 }
90
91 {
92 std::string string_val;
93 // If it cannot pin the value, it copies the value to its internal buffer.
94 // The intenral buffer could be set during construction.
95 PinnableSlice pinnable_val(&string_val);
96 db->Get(ReadOptions(), db->DefaultColumnFamily(), "key2", &pinnable_val);
97 assert(pinnable_val == "value");
98 // If the value is not pinned, the internal buffer must have the value.
99 assert(pinnable_val.IsPinned() || string_val == "value");
100 }
101
102 PinnableSlice pinnable_val;
103 s = db->Get(ReadOptions(), db->DefaultColumnFamily(), "key1", &pinnable_val);
104 assert(s.IsNotFound());
105 // Reset PinnableSlice after each use and before each reuse
106 pinnable_val.Reset();
107 db->Get(ReadOptions(), db->DefaultColumnFamily(), "key2", &pinnable_val);
108 assert(pinnable_val == "value");
109 pinnable_val.Reset();
110 // The Slice pointed by pinnable_val is not valid after this point
111
112 delete db;
113 LOG(ERROR) << "sample done";
114
115 return 0;
116}
