85// Connect to the server if not connected and send data.
86int ResDBClient::SendFromKeepAlive(const std::string& data) {
87 for (int i = 0; i < max_retry_time_; ++i) {
88 if (!long_connecting_) {
89 if (Connect() == 0) {
90 long_connecting_ = true;
91 }
92 if (!long_connecting_) {
93 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
94 continue;
95 }
96 }
97 int ret = SendDataInternal(data);
98 if (ret >= 0) {
99 return ret;
100 }
101 long_connecting_ = false;
102 }
103 return -1;
104}

106int ResDBClient::Send(const std::string& data) {
107 if (long_connect_tion_) {
108 return SendFromKeepAlive(data);
109 }
110 for (int i = 0; i < max_retry_time_; ++i) {
111 if (!connected_) {
112 if (Connect()) {
113 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
114 continue;
115 }
116 }
117 int ret = SendDataInternal(data);
118 if (ret >= 0) {
119 return ret;
120 } else if (ret == -1) {
121 LOG(ERROR) << "send data fail:" << ip_ << " port:" << port_;
122 }
123 }
124 return -1;
125}

137// Sign the message if verifier has been provied and send the message to the
138// server.
139std::string ResDBClient::GetRawMessageString(
140 const google::protobuf::Message& message, SignatureVerifier* verifier) {
141 ResDBMessage sig_message;
142 if (!message.SerializeToString(sig_message.mutable_data())) {
143 return "";
144 }
145
146 if (verifier != nullptr) {
147 auto signature_or = verifier->SignMessage(sig_message.data());
148 if (!signature_or.ok()) {
149 LOG(ERROR) << "Sign message fail";
150 return "";
151 }
152 sig_message.mutable_signature()->Swap(&(*signature_or));
153 } else {
154 // LOG(ERROR) << " no verifier";
155 }
156
157 std::string message_str;
158 if (!sig_message.SerializeToString(&message_str)) {
159 return "";
160 }
161 return message_str;
162}

172int ResDBClient::SendRawMessage(const google::protobuf::Message& message) {
173 ResDBMessage sig_message;
174 if (!message.SerializeToString(sig_message.mutable_data())) {
175 return -1;
176 }
177
178 if (verifier_ != nullptr) {
179 auto signature_or = verifier_->SignMessage(sig_message.data());
180 if (!signature_or.ok()) {
181 LOG(ERROR) << "Sign message fail";
182 return -1;
183 }
184 sig_message.mutable_signature()->Swap(&(*signature_or));
185 }
186 std::string message_str;
187 if (!sig_message.SerializeToString(&message_str)) {
188 return -1;
189 }
190 return Send(message_str);

220 LOG(ERROR) << "parse response msg fail";
