45// Obtain ReplicaState of each replica.
46absl::StatusOr<std::vector<std::pair<uint64_t, std::string>>>
47ResDBTxnClient::GetTxn(uint64_t min_seq, uint64_t max_seq) {
48 QueryRequest request;
49 request.set_min_seq(min_seq);
50 request.set_max_seq(max_seq);
51
52 std::vector<std::unique_ptr<ResDBClient>> clients;
53 std::vector<std::thread> ths;
54 std::string final_str;
55 std::mutex mtx;
56 std::condition_variable resp_cv;
57 bool success = false;
58 std::map<std::string, int> recv_count;
59 for (const auto& replica : replicas_) {
60 std::unique_ptr<ResDBClient> client =
61 GetResDBClient(replica.ip(), replica.port());
62 ResDBClient* client_ptr = client.get();
63 clients.push_back(std::move(client));
64
65 ths.push_back(std::thread(
66 [&](ResDBClient* client) {
67 std::string response_str;
68 int ret = client->SendRequest(request, Request::TYPE_QUERY);
69 if (ret) {
70 return;
71 }
72 client->SetRecvTimeout(1000);
73 ret = client->RecvRawMessageStr(&response_str);
74 if (ret == 0) {
75 std::unique_lock<std::mutex> lck(mtx);
76 recv_count[response_str]++;
77 // receive f+1 count.
78 if (recv_count[response_str] == config_.GetMinClientReceiveNum()) {
79 final_str = response_str;
80 success = true;
81 // notify the main thread.
82 resp_cv.notify_all();
83 }
84 }
85 return;
86 },
87 client_ptr));
88 }
89
90 {
91 std::unique_lock<std::mutex> lck(mtx);
92 resp_cv.wait_for(lck, std::chrono::seconds(recv_timeout_));
93 // Time out or done, close all the client.
94 for (auto& client : clients) {
95 client->Close();
96 }
97 }
98
99 // wait for all theads done.
100 for (auto& th : ths) {
101 if (th.joinable()) {
102 th.join();
103 }
104 }
105
106 std::vector<std::pair<uint64_t, std::string>> txn_resp;
107 QueryResponse resp;
108 if (success && final_str.empty()) {
109 return txn_resp;
110 }
111
112 if (final_str.empty() || !resp.ParseFromString(final_str)) {
113 LOG(ERROR) << "parse fail len:" << final_str.size();
114 return absl::InternalError("recv data fail.");
115 }
116 for (auto& transaction : resp.transactions()) {
117 txn_resp.push_back(std::make_pair(transaction.seq(), transaction.data()));
118 }
119 return txn_resp;
120}
