49int TransactionConsensor::SaveResult(std::unique_ptr<Request> request){
50 int received_size = 0;
51 LOG(ERROR)<<"receive result";
52 std::unique_ptr<BlockMiningInfo> mining_result = std::make_unique<BlockMiningInfo>();
53
54 if(!mining_result->ParseFromString(request->data())){
55 LOG(ERROR)<<"parse data fail";
56 return -2;
57 }
58
59 uint64_t seq = mining_result->header().min_seq();
60 std::string hash = GetHashDigest(mining_result->hash());
61 {
62 std::unique_lock<std::mutex> lck(mutex_);
63 results_[std::make_pair(seq, hash)].push_back(std::move(mining_result));
64 received_size = results_[std::make_pair(seq, hash)].size();
65 }
66 LOG(ERROR)<<"receive seq:"<<seq<<" size:"<<received_size;
67
68 if(received_size != 3){
69 return 0;
70 }
71
72 BatchClientRequest batch_request;
73 batch_request.set_ex_data(request->data());
74 batch_request.set_system_data(true);
75
76 auto new_request =
77 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
78
79 batch_request.SerializeToString(new_request->mutable_data());
80 if(verifier_) {
81 auto signature_or = verifier_->SignMessage(new_request->data());
82 if (!signature_or.ok()) {
83 LOG(ERROR) << "Sign message fail";
84 return -2;
85 }
86 *new_request->mutable_data_signature() = *signature_or;
87 LOG(ERROR)<<"sign";
88 }
89
90 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
91 new_request->set_proxy_id(config_.GetSelfInfo().id());
92
93 return ConsensusServicePBFT::SelfPropose(std::move(new_request));
94}
